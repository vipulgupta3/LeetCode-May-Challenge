class Solution {
public:
    int majorityElement(vector<int>& nums) {
        if(nums.size()==0)
            return 0;
        int ans=nums[0];
        int count=1;
        for(int i=1;i<nums.size();i++)
        {
            if(ans==nums[i])
                count++;
            else
                count--;
            if(count<0)
            {ans=nums[i];
            count=0;
        }
        }
        return ans;
    }
};
